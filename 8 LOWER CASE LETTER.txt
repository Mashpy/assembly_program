;EXAMPLE 08: A CASE CONVERSION PROGRAM. ENTER A LOWER CASE LETTER AND WRITE IT IN UPPER CASE.

.MODEL SMALL
.STACK 100H
.DATA
MSG1 DB,'ENTER A LOWER CASE LETTER:$'
MSG2 DB,0DH,0AH,'IN UPPERCASE. IT IS:'
CHAR DB ?'$'
.CODE
MAIN PROC
MOV AX,@DATA
MOV DS,AX
LEA DX,MSG1
MOV AH,9H
INT 21H
MOV AH,1
INT 21H
SUB AL,20H
MOV CHAR,AL
LEA DX,MSG2
MOV AH,09H
INT 21H
MOV AH,4CH
INT 21H
MIAN ENDP
END MAIN
