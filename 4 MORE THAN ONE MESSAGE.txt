;EXAMPLE 04: TO DISPLAY MORE THAN ONE MESSAGE IN DIFFERENT LINES.

TITLE EX_04: DISPLAY MORE THAN ONE MESSAGE.

.MODEL SMALL
.STACK 100H
.DATA 
MSG1 DB'CSTE DEPARTMENTS$'
MSG2 DB 0DH,0AH,'NSTU$'
.CODE
 MAIN PROC
 MOV AX,@DATA
 MOV DS,AX
LEA DX,MSG1
MOV AH,9H
INT 21H
LEA DX,MSG2
MOV AH,9H
INT 21H
MOV AH,4CH
INT 21H
MAIN ENDP
END MAIN
