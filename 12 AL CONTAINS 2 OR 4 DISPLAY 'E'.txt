; IF AL CONTAIN 1 OR 3, DISPLAY '0'. IF AL CONTAINS 2 OR 4 DISPLAY 'E'.

SOLUTION:


.MODEL SMALL
.STACK 100H
.CODE
CMP AL,1
JE ODD
CMP AL,3
JE ODD
CMP AL,2
JE EVEN
CMP AL,4
JE EVEN
JMP EXIT
ODD:
  MOV DL,'0'
  JMP DISPLAY
EVEN:
  MOV DL,'E'
  JMP DISPLAY
DISPLAY:
  MOV AH,2
  INT 21H
EXIT:
  MOV AH,4CH
  INT 21H
  MAIN ENDP
	END MAIN
