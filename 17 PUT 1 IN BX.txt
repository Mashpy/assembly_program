; IF AX CONTAINS A NEGATIVE NUMBER, PUT -1 IN BX, IF AX CONTAINS 0, PUT 0 IN BX; IF AX CONTAINS A POSITIVE NUMBER, PUT 1 IN BX.

SOLUTION: 
.MODEL SMALL
.STACK 100H
.DATA
.CODE
  MAIN PROC
  CMP AX,0
  JL NEGATIVE
  JE ZERO
  JG POSITIVE
NEGATIVE:
MOV BX,-1
JMP DISPLAY
ZERO:
  MOV BX,0
JMP DISPLAY
 POSITIVE:
  MOV DL,'1'
DISPLAY:
  MOV AH,4
  INT 21H
MAIN ENDP
  END MAIN
